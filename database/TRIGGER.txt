DO NOT ALLOW ANY CHANGE TO ITEM RATE IN SUCH A WAY DIFFERENCE IS MORE THAN 25% OF THE EXISTING RATE.
 
CREATE OR REPLACE TRIGGER  TRGDIFFbooks
BEFORE UPDATE
ON books
FOR EACH ROW
DECLARE
    V_DIFF NUMBER(5);
BEGIN
       V_DIFF := ABS(:NEW.PRICE - :OLD.PRICE);

       IF  V_DIFF > :OLD.PRICE * 0.25 THEN
           RAISE_APPLICATION_ERROR(-20014,'INVALID RATE FOR AMOUNT. CHANGE IS TOO BIG');
       END IF;

END;
/
